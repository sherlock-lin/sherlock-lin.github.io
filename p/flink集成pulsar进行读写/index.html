public class PulsarExample {

    public static String serverUrl = "http://localhost:8081";

    public static void main(String[] args) throws Exception {

        /*1. 创建环境变量*/
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        //Flink相关配置(非必要)
//        env.disableOperatorChaining();
//        env.enableCheckpointing(60 * 1000 * 5, CheckpointingMode.EXACTLY_ONCE);
//        env.getCheckpointConfig().setCheckpointTimeout(60000);
//        env.setStateBackend(new MemoryStateBackend());
//        env.setRestartStrategy(RestartStrategies.fixedDelayRestart(0, 3 * 1000));

        /*2. 从Pulsar中读取数据*/
        PulsarSource<String> source = PulsarSource.builder()
                .setServiceUrl(serverUrl)
                .setStartCursor(StartCursor.earliest())
                .setTopics("public/sherlock-namespace-1/my-topic")
                .setDeserializationSchema(new SimpleStringSchema())
                .setSubscriptionName("my-subscription-1")
                .build();

        DataStreamSource<String> streamSource = env.fromSource(source, WatermarkStrategy.noWatermarks(), "Pulsar "
                + "Source");

        /*3. 打印数据*/
        streamSource.print();

        /*4. 启动任务*/
        env.execute();
    }
}

public class PulsarSinkExample {

    public static String serverUrl = "http://localhost:8081";
    public static String adminUrl = "pulsar://localhost:6650";

    public static void main(String[] args) throws Exception {
        /*1. 创建环境变量*/
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        //Flink相关配置(非必要)
//        env.disableOperatorChaining();
//        env.enableCheckpointing(60*1000*5, CheckpointingMode.EXACTLY_ONCE);
//        env.getCheckpointConfig().setCheckpointTimeout(60000);
//        env.setStateBackend(new MemoryStateBackend());
//        env.setRestartStrategy(RestartStrategies.fixedDelayRestart(0, 3 * 1000));

        /*2. 从Pulsar中读取数据*/
        PulsarSource<String> source = PulsarSource.builder()
                .setServiceUrl(serverUrl)
                .setStartCursor(StartCursor.earliest())
                .setTopics("public/sherlock-namespace-1/my-topic")
                .setDeserializationSchema(new SimpleStringSchema())
                .setSubscriptionName("my-subscription-1")
                .build();

        DataStreamSource<String> streamSource = env.fromSource(source, WatermarkStrategy.noWatermarks(), "Pulsar "
                + "Source");

        //定义输出的topic
        String topic = "persistent://public/sherlock-namespace-1/my-topic-output";

        /*3. 写到Pulsar中*/
        PulsarSink<String> sink = PulsarSink.builder()
                .setServiceUrl(serverUrl)
                .setAdminUrl(adminUrl)
                .setTopics(topic)
                .setSerializationSchema(new SimpleStringSchema())
                .setDeliveryGuarantee(DeliveryGuarantee.AT_LEAST_ONCE)
                .build();

        streamSource.sinkTo(sink);

        /*4. 启动任务*/
        env.execute();
    }
}

    public static void commonConsumer() throws Exception {
        final String topic = "persistent://public/sherlock-namespace-1/my-topic-output";

        PulsarClient pulsarClient = PulsarClient.builder().serviceUrl(serverUrl).build();

        Consumer consumer =
                pulsarClient.newConsumer(Schema.STRING).topic(topic).subscriptionType(SubscriptionType.Shared).subscriptionName("sub_07").subscribe();

        while (true) {
            Message<String> message = consumer.receive();
            String msg = message.getValue();
            System.out.println("消息数据为：" + msg);
            consumer.acknowledge(message);
        }
    }
