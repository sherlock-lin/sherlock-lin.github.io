def sampleTopKey( sparkSession: SparkSession, tableName: String, keyColumn: String ): Array[(Int, Row)] = {
  val df: DataFrame = sparkSession.sql("select " + keyColumn + " from " + tableName)
  val top10Key = df
    .select(keyColumn).sample(false, 0.1).rdd // å¯¹keyä¸æ”¾å›é‡‡æ ·
    .map(k => (k, 1)).reduceByKey(_ + _) // ç»Ÿè®¡ä¸åŒkeyå‡ºç°çš„æ¬¡æ•°
    .map(k => (k._2, k._1)).sortByKey(false) // ç»Ÿè®¡çš„keyè¿›è¡Œæ’åº
    .take(10)
  top10Key
}
